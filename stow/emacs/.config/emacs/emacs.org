* Preamble

** Variables

Per-system variables are loaded from =dnl-custom-vars.el= if it exists; otherwise
fallback values are used. This must run before anything else so that font sizes,
agenda paths, etc. are available to the rest of the config.

#+begin_src emacs-lisp
(setq custom-file (expand-file-name "dnl-custom-vars.el" user-emacs-directory))

(cond ((file-exists-p custom-file)
       (load custom-file))
      (t
       (defvar dnl/default-font-size 130)
       (defvar dnl/org-agenda-path "~/Sync/org/")))

(defvar dnl/default-font "Fira Code") ;; sudo dnf install fira-code-fonts
(defvar dnl/indent-width 2)
#+end_src

** Package Management

Bootstrap =package.el= with MELPA and GNU ELPA, then ensure =use-package= is
available. Setting =use-package-always-ensure t= means every =use-package= form
will automatically install the package if it's missing.

#+begin_src emacs-lisp
(require 'package)

(setq package-archives '(("melpa" . "https://melpa.org/packages/")
                         ("elpa"  . "https://elpa.gnu.org/packages/")))

(package-initialize)
(unless package-archive-contents
  (package-refresh-contents))

(require 'use-package)
(setq use-package-always-ensure t)
#+end_src

** Theme

Load the custom Calcite theme from the local =themes/= directory. =auto-dark=
switches between light and dark variants based on the OS appearance setting.

On Linux, Emacs 30+ reads the GTK3 theme. To get automatic switching, install
the [[https://extensions.gnome.org/extension/4998/legacy-gtk3-theme-scheme-auto-switcher/][Legacy GTK3 Switcher Extension]] plus =adw-gtk3-theme= and configure it in
GNOME Tweaks.

#+begin_src emacs-lisp
(add-to-list 'custom-theme-load-path (expand-file-name "themes/" user-emacs-directory))
(load-theme 'calcite-light t)

(defun dnl/reload-theme ()
  "Reload the current theme without confirmation."
  (interactive)
  (let ((current-themes custom-enabled-themes))
    (mapc #'disable-theme custom-enabled-themes)
    (mapc (lambda (theme) (load-theme theme t)) current-themes)))

(use-package auto-dark
  :custom
  (auto-dark-themes '((calcite-dark) (calcite-light)))
  (auto-dark-polling-interval-seconds 5)
  (auto-dark-allow-osascript nil)
  (auto-dark-allow-powershell nil)
  :init (auto-dark-mode))

(use-package nerd-icons)
#+end_src

* Completion & Minibuffer

The completion stack is built around the standard Emacs =completing-read= API:
Vertico for vertical display, Orderless for fuzzy matching, Consult for search
and navigation commands, Marginalia for annotations, Embark for contextual
actions, and Corfu for in-buffer completion.

  :ensure nil
  :bind (("C-c a" . org-agenda)
         ("C-c c" . org-capture))
  :custom
  (org-ellipsis " ▾")
  (org-todo-keywords '((sequence "TODO(t)" "DOING(n)" "BLOCKED(b)" "|" "DONE(d)" "DELEGATED(g)" "CANCELLED(c)")))
  (org-agenda-start-on-weekday 1)
  (calendar-week-start-day 1)
  (org-agenda-skip-scheduled-if-done t)
  (org-log-done t)
  (org-agenda-files `(,dnl/org-agenda-path))
  (org-src-fontify-natively t)
  (org-src-tab-acts-natively t)
  (org-confirm-babel-evaluate nil)
  (org-edit-src-content-indentation 0)
  (org-blank-before-new-entry '((heading . nil) (plain-list-item . nil)))
  :config
  (unless (bound-and-true-p org-capture-templates)
    (setq org-capture-templates
          '(("i" "Inbox" entry (file+headline "~/Sync/org/notes.org" "Inbox")
             "** %?")
            ("t" "Todo" entry (file+headline "~/Sync/org/notes.org" "GTD")

** diff-hl

Show per-line diff indicators in the fringe. We hook into focus and window

  (add-hook 'magit-post-refresh-hook 'diff-hl-magit-post-refresh)
  (add-hook 'after-revert-hook       'diff-hl-update)

  (add-function :after after-focus-change-function
                (lambda ()
                  (when (frame-focus-state)
                    (when diff-hl-mode (diff-hl-update)))))

  (add-hook 'window-selection-change-functions
            (lambda (_)
              (when diff-hl-mode
                (diff-hl-update)))))

** exec-path-from-shell

On macOS, GUI Emacs does not inherit the full shell =PATH=. This package reads

vterm is a fast terminal emulator backed by a native C library. Requires
=cmake= and =libtool= to compile. See the [[https://github.com/akermu/emacs-libvterm#shell-side-configuration][shell-side configuration]] docs for
integrating with your shell.

#+begin_src emacs-lisp
(use-package vterm
  :commands vterm-mode
  :bind
  (("C-c t t" . vterm))
  :config
  (setq vterm-kill-buffer-on-exit t
        vterm-shell (or (executable-find "fish") "/bin/sh")))
#+end_src

** Delta Green Export

Custom LaTeX export for Delta Green TTRPG scenarios. Registers the
=deltagreen= document class and maps org special blocks (=#+begin_scene=,
=#+begin_npc=, etc.) to their corresponding LaTeX environments.

#+begin_src emacs-lisp
(with-eval-after-load 'ox-latex

  (add-to-list 'org-latex-classes
    '("deltagreen"
      "\\documentclass{deltagreen}"
      ("\\section{%s}"       . "\\section*{%s}")
      ("\\subsection{%s}"    . "\\subsection*{%s}")
      ("\\subsubsection{%s}" . "\\subsubsection*{%s}")))

  (setq org-latex-pdf-process
        '("pdflatex -interaction nonstopmode -output-directory %o %f"
          "pdflatex -interaction nonstopmode -output-directory %o %f"))

  (defun dg/latex-special-block (orig-fn special-block contents info)
    "Transcode Delta Green special blocks; fall back to ORIG-FN for others."
    (let* ((type  (downcase (org-element-property :type special-block)))
           (param (org-element-property :parameters special-block)))
      (pcase type
        ("handler"  (format "\\begin{handlerbox}\n%s\\end{handlerbox}\n" contents))
        ("hook"     (format "\\begin{hook}\n%s\\end{hook}\n" contents))
        ("scene"    (format "\\begin{scene}{%s}\n%s\\end{scene}\n"
                            (or param "Scene") contents))
        ("npc"      (format "\\begin{npcblock}{%s}\n%s\\end{npcblock}\n"
                            (or param "NPC") contents))
        ("location" (format "\\begin{locationblock}{%s}\n%s\\end{locationblock}\n"
                            (or param "Location") contents))
        ("cluebox"  (format "\\begin{cluebox}[%s]\n%s\\end{cluebox}\n"
                            (or param "Clue") contents))
        (_          (funcall orig-fn special-block contents info)))))

  (advice-add 'org-latex-special-block :around #'dg/latex-special-block))
#+end_src
